building_Doge_matter_gen = {
	
	base_buildtime = @b3_time

	base_cap_amount = 1

	category = manufacturing

	potential = {
		exists = owner
		OWNER = {
			OR = {
				has_monthly_income = {
					resource = exotic_gases
					value > 0
				}
				AND = {
					is_ai = no
					has_resource = {
						type = exotic_gases
						amount > 0
					}
				}
			}
		}
		building_basic_income_check = yes
	}

	allow = {
		has_upgraded_capital = yes
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 1200
			exotic_gases = 100
		}
		upkeep = {
			energy = 20
		}
	}

	prerequisites = { "tech_Doge_matter_enrich" }

	triggered_planet_modifier = {
		potential = {
				exists = owner
				owner = {
					is_regular_empire = yes
				}
		}

		modifier = {
			job_Doge_matter_generator_add = 2
		}
	}

	triggered_planet_modifier = {
        potential = {
	    exists = owner
            owner = { is_gestalt = yes }
        }
        modifier = {
            job_Doge_matter_generator_drone_add = 2
        }
    }

	upgrades = {
		building_Doge_matter_gen_t2
	}

	ai_resource_production = {
		trigger = { always = yes }
		sr_Doge_matter = 4
	}
}
